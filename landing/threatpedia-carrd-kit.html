STEP 8 — EMBED #8: JAVASCRIPT (Hidden, Body End)

In Carrd: Add Element → Embed → Type: Code → Style: Hidden → Location: Body End

<script>
(function() {
  var form = document.getElementById('tp-form');
  var success = document.getElementById('tp-success');
  if (!form) return;

  form.addEventListener('submit', function(e) {
    e.preventDefault();
    var data = new FormData(form);
    fetch(form.action, {
      method: 'POST',
      body: data,
      headers: { 'Accept': 'application/json' }
    })
    .then(function(res) {
      if (res.ok) {
        form.style.display = 'none';
        success.classList.add('tp-show');
      } else {
        alert('Something went wrong. Please try again.');
      }
    })
    .catch(function() {
      alert('Network error. Please try again.');
    });
  });
})();
</script>
